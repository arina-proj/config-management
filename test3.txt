# Тестирование ЗАГРУЗКИ VFS и базовых команд

# ==================== ТЕСТИРОВАНИЕ ЗАГРУЗКИ VFS ====================

# 1. Проверка содержимого загруженной VFS
ls

# 2. Навигация по структуре (требование: 3+ уровня)
cd level1
ls
cd level2
ls  
cd level3
ls
cd ../../..

# ==================== ТЕСТИРОВАНИЕ БАЗОВЫХ КОМАНД ====================

# 3. Тестирование ls с разными аргументами
ls -l
ls -a
ls /nonexistent

# 4. Тестирование cd с разными путями
cd /absolute/path
cd relative/path
cd .

# ==================== ТЕСТИРОВАНИЕ ОШИБОК ====================

# 5. Ошибки навигации
cd nonexistent_directory
ls /invalid/path

# 6. Несуществующие команды
invalid_command
mkdir  # НЕ РАБОТАЕТ - по заданию не нужно!
touch  # НЕ РАБОТАЕТ - по заданию не нужно!
rm     # НЕ РАБОТАЕТ - по заданию не нужно!

# 7. Специальные случаи
cd "path with spaces"
ls --invalid-option

# ==================== КОМПЛЕКСНЫЕ СЦЕНАРИИ ====================

# 8. Последовательность команд
ls
cd level1
ls
cd ../level2_alternative
ls
cd ..

# 9. Тестирование после ошибок
ls  # должен работать после предыдущих ошибок

# ==================== ФИНАЛЬНОЕ ТЕСТИРОВАНИЕ ====================

# 10. Завершение работы
# exit

echo "Тестирование загрузки VFS завершено"
